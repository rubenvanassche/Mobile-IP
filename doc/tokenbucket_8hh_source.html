208 {

210  rate = 0;
211  capacity = max_tokens;
212  }

214  token_type frequency = P::frequency();
215  if (rate != 0) {
216  // constrain capacity so _tokens_per_tick fits in 1 limb
217  unsigned min_capacity = (rate - 1) / frequency + 1;
218  if (capacity < min_capacity)
219  capacity = min_capacity;
220  }

225  "bigint::limb_type should have the same size as token_type.");

229  (void) bigint::divide(l, l, 2, frequency);

579 template <typename R>
580 void TokenCounterX<R>::refill(const rate_type &rate, time_point_type time)
581 {
582  ticks_type diff = rate.time_monotonic_difference(_time_point, time);
583  if (diff >= rate.time_until_full()) {
584  // ignore special case of idle rates -- we assume that
585  // rate.time_monotonic_difference() will never return (ticks_type) -1,
586  // and ensure that ticks_type is unsigned
587  _tokens = max_tokens;
588  } else if (diff > 0) {
589  token_type new_tokens = _tokens + diff * rate.tokens_per_tick();
590  _tokens = (new_tokens < _tokens ? (token_type) max_tokens : new_tokens);
591  }
592  _time_point = time;
593 }
