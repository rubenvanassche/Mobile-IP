46 /*
47  * IEEE 802.2 Link Level Control headers, for use in conjunction with
48  * 802.{3,4,5} media access control methods.
49  *
50  * Headers here do not use bit fields due to shortcomings in many
51  * compilers.
52  */
53 
54 struct click_llc {
55  uint8_t llc_dsap;
56  uint8_t llc_ssap;
57  union {
58  struct {
59  uint8_t control;
60  uint8_t format_id;
61  uint8_t _class;
62  uint8_t window_x2;
63  } type_u;
64  struct {
65  uint8_t num_snd_x2;
66  uint8_t num_rcv_x2;
67  } type_i;
68  struct {
69  uint8_t control;
70  uint8_t num_rcv_x2;
71  } type_s;
72  struct {
73  uint8_t control;
74  struct frmrinfo {
75  uint8_t rej_pdu_0;
76  uint8_t rej_pdu_1;
77  uint8_t frmr_control;
78  uint8_t frmr_control_ext;
79  uint8_t frmr_cause;
80  } frmrinfo;
81  } type_frmr;
82  struct {
83  uint8_t control;
84  uint8_t org_code[3];
85  uint16_t ether_type;
86  } type_snap;
87  struct {
88  uint8_t control;
89  uint8_t control_ext;
90  } type_raw;
91  } llc_un;
92 } CLICK_SIZE_PACKED_ATTRIBUTE;
