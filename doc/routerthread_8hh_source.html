401 inline void
402 RouterThread::block_tasks(bool scheduled)
403 {
404  assert(!current_thread_is_running() && !running_in_interrupt());
405  if (!scheduled)
406  ++_task_blocker_waiting;
407  while (1) {
408  uint32_t blocker = _task_blocker.value();
409  if ((int32_t) blocker >= 0
410  && _task_blocker.compare_swap(blocker, blocker + 1) == blocker)
411  break;
412 #if CLICK_LINUXMODULE
413  // 3.Nov.2008: Must allow other threads a chance to run. Otherwise,
414  // soft lock is possible: the thread in block_tasks() waits for
415  // RouterThread::_linux_task to complete a task set, but
416  // RouterThread::_linux_task can't run until the thread in
417  // block_tasks() relinquishes the CPU.
418  //
419  // We might be able to avoid schedule() in some cases, but don't
420  // bother to try.
421  schedule();
422 #endif
423  }
424  --_task_blocker_waiting;
425 }
