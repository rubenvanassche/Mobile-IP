72 static inline int md5_init(md5_state_t *pms) {
73 #if LINUX_VERSION_CODE >= KERNEL_VERSION(2,6,19)
74  pms->tfm = crypto_alloc_hash("md5", 0, CRYPTO_ALG_ASYNC);
75  if (IS_ERR(pms->tfm))
76  return PTR_ERR(pms->tfm);
77  pms->flags = 0;
78  crypto_hash_init(pms);
79  return 0;
80 #else
81  *pms = crypto_alloc_tfm("md5", 0);
82  if (IS_ERR(*pms))
83  return PTR_ERR(*pms);
84  crypto_digest_init(*pms);
85  return 0;
86 #endif
87 }

97 static inline void md5_append(md5_state_t *pms, const unsigned char *data, int nbytes) {
98  struct scatterlist sg;
99  if (likely(virt_addr_valid((unsigned long)data))) {
100  sg_init_one(&sg, const_cast<uint8_t *>(data), nbytes);
101 #if LINUX_VERSION_CODE >= KERNEL_VERSION(2,6,19)
102  crypto_hash_update(pms, &sg, nbytes);
103 #else
104  crypto_digest_update(*pms, &sg, 1);
105 #endif
106  } else {
107  while (nbytes > 0) {
108  int len = nbytes;
109  int off = offset_in_page(data);
110  if (off + len > (int)PAGE_SIZE)
111  len = PAGE_SIZE - off;
112  sg_init_table(&sg, 1);
113  sg_set_page(&sg, vmalloc_to_page(const_cast<uint8_t *>(data)), len, off);
114 #if LINUX_VERSION_CODE >= KERNEL_VERSION(2,6,19)
115  crypto_hash_update(pms, &sg, len);
116 #else
117  crypto_digest_update(*pms, &sg, 1);
118 #endif
119  data += len;
120  nbytes -= len;
121  }
122  }
123 }

143 /*
144  * This package supports both compile-time and run-time determination of CPU
145  * byte order. If ARCH_IS_BIG_ENDIAN is defined as 0, the code will be
146  * compiled to run only on little-endian CPUs; if ARCH_IS_BIG_ENDIAN is
147  * defined as non-zero, the code will be compiled to run only on big-endian
148  * CPUs; if ARCH_IS_BIG_ENDIAN is not defined, the code will be compiled to
149  * run on either big- or little-endian CPUs, but will run slightly less
150  * efficiently on either one than if ARCH_IS_BIG_ENDIAN is defined.
151  */
